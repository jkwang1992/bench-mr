250  unsigned int rounds, double eta,
251  double discount = 1.) {
252  double dx, dy;
253  for (int round = 0; round < rounds; ++round) {
254  // gradient descent along distance field, excluding start/end nodes
255  for (int i = 1; i < path.getStateCount() - 1; ++i) {
256  // compute gradient
257  global::settings.environment->distanceGradient(
258  path.getStates()[i]->as<State>()->getX(),

260  double distance = global::settings.environment->bilinearDistance(

263  distance = std::max(.1, distance);
264  path.getStates()[i]->as<State>()->setX(

349  static std::vector<Point> equidistantSampling(const std::vector<Point> &path,
350  size_t targetSize) {
351  std::vector<Point> result;
352  const double total = totalLength(path);
353  const double targetSegment = total / targetSize;
354  result.emplace_back(path[0]);
355  double sofar = 0;
356  double segment = 0;
357  size_t resultCounter = 1;
358  for (size_t i = 1; i < path.size(); ++i) {
359  double dx = (path[i].x - path[i - 1].x);
360  double dy = (path[i].y - path[i - 1].y);
361  const double l = std::sqrt(dx * dx + dy * dy);
362  if (std::abs(l) < 1e-3) continue;
363  dx /= l;
364  dy /= l;
365  if (segment + l < targetSegment) {
366  segment += l;
367  OMPL_DEBUG("EquidistantSampling: Segment too short between %i and %i",
368  i - 1, i);
369  continue;
370  }
371  double start = std::fmod(segment + l, targetSegment) - segment;
372  segment = 0;
373  const auto segmentSteps =
374  static_cast<unsigned int>(std::floor((segment + l) / targetSegment));
375  for (unsigned int j = 0; j < segmentSteps; ++j) {
376  const double alpha = start + j * targetSegment;
377  const Point p(path[i - 1].x + dx * alpha, path[i - 1].y + dy * alpha);
378  result.emplace_back(p);
379  }
380  sofar += segmentSteps * targetSegment;
381  segment = l - segmentSteps * targetSegment;
382  }
383  result.emplace_back(path.back());
384  return result;
385  }
